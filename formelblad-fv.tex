\begin{itemize}

\item %% Partial derivatives for a function of two variables at a point $(a,b)$:%<<<
  \Tr{Partial derivatives for a function of two variables at a point}
     {Partiella derivator för funktion av två variabler i punkten}
$(a,b)$:
 %
 $$
%%  \eqalign{
%%  \dd fx(a,b)&=D_xf(a,b)=D_1f(a,b)=f'_x(a,b)=\lim_{h\to0}\frac{f(a+h,b)-f(a,b)}{h}\cr

 $$

\item % The gradient of %<<<
  \Tr{The gradient of}
     {Gradiententen av}
  $f:\Rone^n\to\Rone$: $\mbox{grad}\, f(\bdx)
  =\nabla f(\bdx)
  =(\dd f{x_1}(\bdx), \ldots, \dd f{x_n}(\bdx))^T
  =(f'_{x_1}(\bdx),\ldots, f'_{x_n}(\bdx))^T
  $.%>>>

\item %% Diff-bility %<<<
  \Tr{A function}
     {Funktionen}
  $f:\Rone^n\to\Rone$
  \Tr{is differentiable at a point}
     {är differentierbar i punkten}
  $\bda=(a_1,\cdots,a_n)$
  \Tr{if}
     {om}
  $$
    f(\bda+\bdh) = f(\bda)+\nabla f(\bda)\cdot\bdh + |\bdh|\rho(\bdh)
  $$
  \Tr{in a neighbourhood of}
     {i någon omgivning av}
  $\bda$,
  \Tr{so that}
     {så att}
  $\rho(\bdh)\to0$,
  \Tr{as}
     {då}
  $\bdh\to\bdzero$.
  \Tr{For differentiable function of two variables:}
     {För differentierbara funktioner av två variabler:}
  $$
  f(a+h,b+k)
  = f(a,b)+f'_x(a,b)(x-a)+f'_y(a,b)(y-b)+\sqrt{h^2+k^2}\,\rho(h,k).
  $$%>>>

\item %% The directional derivative  %<<<
  \Tr{The directional derivative of a differentiable}
     {Riktningsderivatan av en differentierbar}
$f:\Rone^n\to\Rone$
\Tr{in the direction}
   {i riktningen}
$\bdv$, $|\bdv|=1$,
\Tr{at the point}
   {i punkten}
$\bda$:
%
 $$
 \dd f{\bdv}(a,b)=f'_{\bdv}(a,b)=\lim_{t\to0}\frac{f(\bda+t\bdv)-f(\bda)}{t}
                 =\nabla f(\bda) \cdot \bdv.
 $$
 % >>>

\item % Tangentplan till funktionsyta %<<<
  \Tr{Tangent plane}
     {Tangentplan}
  $\Pi$
  \Tr{and normal line}
     {och normallinje}
  $N$
  \Tr{to the functional surface}
     {till funktionsytan}
  $z=f(x,y)$
  \Tr{at the point}
     {i punkten}
  $(a,b)$:
  $$
  \Pi: z = f(a,b)+f'_x(a,b)(x-a)+f'_y(a,b)(y-b); \qquad
  N: \trevektor[x,y,z]=\trevektor[a,b,f(a,b)] + t\pmatrix{f'_x(a,b)\cr f'_y(a,b)\cr -1},\; t\in\Rone.
  $$%>>>

\item % Tangentplan till F(x,y,z) = 0 %<<<
  \Tr{Tangent plane}
     {Tangentplan}
  $\Pi$
  \Tr{and normal line}
     {och normallinje}
  $N$
  \Tr{to the implicitly given surface}
     {till implicitangivna ytan}
  $F(x,y,z)=0$
  \Tr{at the point}
     {i punkten}
  $(a,b,c)$:
  $$
  \Pi: \nabla F(a,b,c)\cdot\trevektor[x-a,y-b,z-c]=0\qquad
  N: \trevektor[x,y,z]=\trevektor[a,b,c] + t\nabla F(a,b,c),\; t\in\Rone.
  $$%>>>

\item % Differential %<<<
  \Tr{The the differential (the total derivative) and the linearization of}
     {Differentialen och lineariseringen av}
$f:\Rone^n\to\Rone$
\Tr{at a point}
   {i punkten}
$\bda$:
  $$
  df(\bda)=\nabla f(\bda)\cdot d\bdx=f'_{x_1}(\bda)dx_1+\cdots+f'_{x_n}(\bda)dx_n;
  \quad
   \Delta f(\bda) =f(\bda+\Delta\bdx)-f(\bda)
                  \approx  \nabla f(\bda)\cdot \Delta \bdx.
                  % =f'_{x_1}(\bda)\Delta x_1+\cdots+f'_{x_n}(\bda)\Delta x_n\cr
  $$%>>>

\item % The chain rule %<<<
  \Tr{The chain rule. If}
     {Kedjeregeln. Om}
  $f:\Rone^n\to\Rone$,
  $\bdr=\bdr(t)=(x_1(t),\dots, x_n(t))$, and $\varphi(t)=f(\bdr(t))$,
  \Tr{then:}
     {då är}
  $$
  \varphi'(t)=\frac{d}{dt} f(\bdr(t))
          =f'_{x_1}(\bdr(t))\,x_1'(t)+\cdots +f'_{x_n}(\bdr(t))\,x_n'(t)
          =\nabla f(\bdr(t))\cdot\bdr'(t).
  $$
  %>>>

  \item % Derivator av högre ordning %<<<
    \Tr{\textbf{The higher-order derivatives} are defined recursively, e.g.}
       {\textbf{Högre ordningens derivator} definieras rekursivt, dvs}
  \ $f''_{xx}\stackrel{\mbox{\tiny def}}=(f'_x)'_x$,
  \ $f''_{xy}\stackrel{\mbox{\tiny def}}=(f'_x)'_y$, \ etc.
A f-n
$f\in C^n(\Omega)$ 
\Tr{if $f$ and all derivatives up to order}
   {om $f$ och alla dess derivator till ordning}
$n$ 
\Tr{are continuous functions at every point of}
   {är kontinuerliga funktioner i alla punkter av}
$\Omega$.%>>>

\item % The mixed derivatives theorem %<<<
  \Tr{The mixed derivatives theorem gives a sufficient condition for equality of the mixed derivatives, e.g.:}
     {Ett tillräckligt villkor för likhet av de blandade derivatorna är:}
  $$
  f:\Rone^2\to\Rone,\, f\in C^2(\Omega) \quad\ergo\quad f''_{xy}=f''_{yx}.
  $$%>>>

\end{itemize}

 $\Delta\stackrel{\mbox{\tiny def}}

\subsection*{
\Tr{Double integrals}
   {Dubbelintegraler}
}

\begin{itemize}

 \item % Iterated integration (Fubini theorem): % <<<
   \Tr{Iterated integration (Fubini theorem):}
      {Itererad integration (Fubinis sats):}
     $$
        \iint_{\mkern-12mu D} f(x,y)\,dxdy
        =\int_a^b\Bigl(\int_{\varphi(x)}^{\psi(x)}f(x,y)\,dy\Bigr)dx,
        \qquad D=\{\varphi(x)\le y \le \psi(x),\, a\le x\le b\}.
     $$
   % >>>

 \item % Change of variables %<<<
   \Tr{Change of variables $D\leftrightarrow D'$ by}
      {Variabelbyte $D\leftrightarrow D'$ genom}
      \ $\cases{x=x(u,v)\cr y=y(u,v)}$ 
      \Tr{or}{eller}
      $\cases{u=u(x,y)\cr v=v(x,y)}$
$$
  \iint_{\mkern-12mu D} f(x,y)\,dxdy =
  \iint_{\mkern-12mu D'} f(x(u,v),y(u,v))\,J\,dudv,
  \qquad
  J=\Bigl|\frac{d(x,y)}{d(u,v)}\Bigr| =
  \frac1{\Bigl|\frac{d(u,v)}{d(x,y)}\Bigr|}\cdot
$$%>>>

  \item % Polära koordinater %<<<
    \Tr{Polar coordinates}
       {Polära koordinater}
       ($a>0$, $b>0$):
       \ $\cases{x=ar\cos\varphi\cr y = br\sin\varphi}$, \hspace{0.5em}
      $dxdy = J\,drd\varphi$, \ $J=\Bigl|\frac{d(x,y)}{d(r,\varphi)}\Bigr|=abr$.
      %>>>

\end{itemize}

\subsection*{
\Tr{Flux integrals}
   {Flödesintegraler}
}% over scalar fields}

\Tr{Notation:}
   {Beteckning:}
$\bdF=(P,Q,R):\Rone^3\to\Rone^3$, $S$
\Tr{is a orientable, piecewise $C^1$-surface, $\widehat\bdN$ is the unit normal to $S$.}
   {är orienterbar, styckvis $C^1$-yta, $\widehat\bdN$ är enhetsnormalen till $S$.}
\begin{itemize}

% Computation %<<<
   \item 
     \Tr{If}
        {Om}
     $S: \bdr=\bdr(u,v)$, $(u,v)\in\Omega\subset\Rone^2$,
     \Tr{then}
        {då är}
     $$
     \iint_{\mskip -12mu S} \bdF\cdot\widehat\bdN\,dS
   = \pm\iint_{\mskip -12mu\Omega}
         \bdF(\bdr(u,v))\cdot (\bdr'_u\times\bdr'_v)\,dudv.
     $$%>>>

%   If $S$ is the functional surface %<<<
   \item 
     \Tr{If, in particular, $S$ is the functional surface}
        {Om, i synnerhet, $S$ är en funktionsyta,}
     $S: z=z(x,y)$, $(x,y)\in\Omega\subset\Rone^2$,
     \Tr{then}
        {då är}
     $$
     \iint_{\mskip -12mu S} \bdF\cdot\widehat\bdN\,dS
   = \pm\iint_{\mskip -12mu\Omega}
     \bdF(x,y,z(x,y))\cdot\trevektor[-z'_x,-z'_y,1]\,dxdy.
     $$%>>>

%  Gauss %<<<
   \item 
     \Tr{The Gauss (divergence) theorem: If}
        {Gauss'  sats (divergenssatsen): Om}
     $V\subset\Rone^3$
     \Tr{has a piecewise $C^1$--boundary}
        {har styckvis $C^1$--rand}
     $S$
     \Tr{with an outer unit normal $\widehat\bdN$ and $\bdF\in C^1(V\cup\partial V)$, then}
        {med enhetsnormal utåt $\widehat\bdN$ och $\bdF\in C^1(V\cup\partial V)$, då är}
     $$
     \oiint_{\mskip -12mu\partial V} \bdF\cdot\widehat\bdN\,dS
   = \iiint_{\mskip -12mu V} \Div\bdF \,dV
   = \iiint_{\mskip -12mu V} \nabla\cdot \bdF \,dV
   = \iiint_{\mskip -12mu V} (P'_x+Q'_y+R'_z) \,dV.
     $$%>>>

%  Stokes %<<<
   \item 
     \Tr{The Stokes theorem:}
        {Stokes' sats:}
      \Tr{If the $C^1$-surface $S$ has a piecewise $C^1$-boundary $\partial S$, then the counterclockwise circulation (seen from}
         {Om $C^1$-ytan $S$ har styckvis $C^1$-rand $\partial S$, då är moturscirkulationen (sett från}
     $\widehat\bdN$)
     $$
     \oint_{\partial S} \bdF\cdot d\bdr
     = \iint_{\mskip -12mu S} \Tr{\Curl}{\Rot}\bdF\cdot\widehat\bdN\,dS
   = \iint_{\mskip -12mu S} (\nabla\times \bdF)\cdot\widehat\bdN\,dS.
     $$
     %>>>
\end{itemize}
